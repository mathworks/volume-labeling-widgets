Isosurface Viewer

Load in data (MRI)

s = load('mri.mat');
data = squeeze(s.D);
% Create isosurface
isovalue = 40;
[faces, vertices] = isosurface(data, isovalue);
voxel_size = [1 1 3];
vertices = vertices .* voxel_size;
vertexNormals = isonormals(data,vertices);
% Place into an IsosurfaceModel
isoModel = wt.model.IsosurfaceModel(...
'Faces',faces,...
'Vertices',vertices,...
'VertexNormals',vertexNormals);

Create the widget

f = uifigure;
g = uigridlayout(f,'ColumnWidth',{'1x'},'RowHeight',{'1x'},'Padding',[0 0 0 0]);
w = wt.IsosurfaceViewer('Parent',g,'IsosurfaceModel',isoModel);
movegui(f,'northwest')
